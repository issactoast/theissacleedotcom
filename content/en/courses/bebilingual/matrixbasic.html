Python setup

The python codes below are written using the NumPy python package, so import NumPy before we run them.

import numpy as np

Define

Method 1

# Python
m = np.array([[4, 6, 9],
              [1, 3, 4],
              [1, 2, 3]])
m              
#result>> array([[4, 6, 9],
#result>>        [1, 3, 4],
#result>>        [1, 2, 3]])
# R
m <- rbind(c(4, 6, 9),
           c(1, 3, 4),
           c(1, 2, 3))
m
#result>>      [,1] [,2] [,3]
#result>> [1,]    4    6    9
#result>> [2,]    1    3    4
#result>> [3,]    1    2    3

Method 2

# Python
m = np.array([np.arange(5), 
              np.arange(5)])
m
#result>> array([[0, 1, 2, 3, 4],
#result>>        [0, 1, 2, 3, 4]])
# R
m <- rbind(0:4, 0:4)
m
#result>>      [,1] [,2] [,3] [,4] [,5]
#result>> [1,]    0    1    2    3    4
#result>> [2,]    0    1    2    3    4

Method 3

# Python
m = np.array([4, 6, 9,
              1, 3, 4, 
              1, 2, 3]).reshape(3, 3)
m
#result>> array([[4, 6, 9],
#result>>        [1, 3, 4],
#result>>        [1, 2, 3]])
# R
m <- matrix(c(4, 6, 9,
              1, 3, 4,
              1, 2, 3), 3, 3, byrow = T)
m
#result>>      [,1] [,2] [,3]
#result>> [1,]    4    6    9
#result>> [2,]    1    3    4
#result>> [3,]    1    2    3

Dimension

# Python
m.shape
#result>> (3, 3)
# R
dim(m)
#result>> [1] 3 3

Access

# Python
m[0, 0]; m[0, 1]; m[1, 0]; m[1, 1]
#result>> 4
#result>> 6
#result>> 1
#result>> 3
# R
m[1, 1]; m[1, 2]; m[2, 1]; m[2, 2]
#result>> [1] 4
#result>> [1] 6
#result>> [1] 1
#result>> [1] 3
